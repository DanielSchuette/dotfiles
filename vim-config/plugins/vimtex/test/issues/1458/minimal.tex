\documentclass[12pt]{article}
\usepackage{pgfplots}

\pgfplotsset{
  compat=1.16,
  3d_style/.append style={
    every non boxed x axis/.append style={x axis line style=-},
    every non boxed y axis/.append style={y axis line style=-},
    axis z line=none,
    axis x line=left,
    axis y line=left,
    xlabel=$x$,
    ylabel=$y$,
    colormap={graykst}{gray(0cm)=(0.35); gray(1cm)=(1)},
    tick label style={font=\small},
    label style={font=\small},
    view={45}{45}
  }
}

\begin{document}

\begin{center}
  \begin{tikzpicture}
    \begin{logaxis}[3d_style, xshift=2cm]
      \addplot3 [surf,samples=15, domain=-2:2,miter limit=1 ] {0.1*exp((-x^2-y^2))};
      \coordinate (x_axis) at (0,0);
    \end{logaxis}
    \node at (5.8,3.5) {$f(x,y)$};
  \end{tikzpicture}
\end{center}

\end{document}
